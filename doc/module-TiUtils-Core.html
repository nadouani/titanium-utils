Example

extract the firstname from a list of persons

 {firstname: 'John', lastname:'Doe'},
 {firstname: 'Chuck', lastname:'Norris'}
];

TiUtils.prototype.pluck(persons, 'firstname');
// returns ['John', 'Chuck']
